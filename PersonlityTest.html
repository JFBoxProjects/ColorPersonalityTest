  <script>

    function calculateResults() {

      const totals = { A: 0, B: 0, C: 0, D: 0 };

      for (let i = 1; i <= 20; i++) {

        totals.A += parseInt(document.querySelector(`input[name="q${i}a"]`).value || 0);

        totals.B += parseInt(document.querySelector(`input[name="q${i}b"]`).value || 0);

        totals.C += parseInt(document.querySelector(`input[name="q${i}c"]`).value || 0);

        totals.D += parseInt(document.querySelector(`input[name="q${i}d"]`).value || 0);

      }


      const sorted = Object.entries(totals).sort((a, b) => b[1] - a[1]);

      const colors = { A: "Blue", B: "Green", C: "Red", D: "Yellow" };


      let resultText = `<strong>Your Personality Ranking:</strong><br>`;

      sorted.forEach(([key, value]) => {

        resultText += `${colors[key]}: ${value}<br>`;

      });


      document.getElementById("result").innerHTML = resultText;

    }
